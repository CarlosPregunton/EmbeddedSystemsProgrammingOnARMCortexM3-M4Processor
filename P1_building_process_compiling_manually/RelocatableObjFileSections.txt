# Execute the following instruction on the shell to see commands
arm-none-eabi-objdump
# This instruction show header sections
arm-none-eabi-objdump -h main.o
# This instruction show the content of all sections
arm-none-eabi-objdump -s main.o
# This instruction disassembly .text section
arm-none-eabi-objdump -d main.o
# This instruction disassembly all sections
arm-none-eabi-objdump -D main.o

# Read Only Memory, (ROM) which could be Flash technology, contains:
# - Code section .text
# - Read only data .rodata
# - Some special sections added by the compiler. Contains some some
#   special data.
# - User defined section (code or data)

# Random Access Memory (RAM)
# - Initialized data .data
# - Contains data which are uninitialized .bss
# - User defined section (code or data)
